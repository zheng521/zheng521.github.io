new 实现

  1. 创建一个新对象
  2. 将新对象的原型属性指向构造函数的原型对象
  3. 将构造函数的作用域赋给新对象（因此this就指向了这个新对象）
  4. 执行构造函数中的代码（为这个新对象添加属性）
  5. 返回新对象 (判断新对象是否为对象)

``

function myNew(Func) {
  let obj = {}
  obj._prop_ = Func.prototype 
  let result = Func.apply(obj, arguments)
  return  result instanceof Object ? result : obj
}

instaceof 实现

function myInstaceOf(left, right) {
  let prop = Object.getPrototypeOf(left)
  let prototype = right.prototype
  while (true) {
    if (prop === null) return false 
    if (prop === prototype) {
      return true
    }
    prop = Object.getPrototypeOf(prop)
  }
}

currry实现

function curry(fn, ...args) {
  if (args.length > fn.length) {
    fn(...args)
  } else {
    return (..._args) => curry(fn, ...args, ..._args)
  }
}

throttle

function throttle(fn, delay) {
  let timer = null
  return function (...args) {
    if (!timer) {
      timer = setTimeout(() => {
        fn.apply(this, args)
        clearTimeout(timer)
      }, delay)
    }
  }
}

debounce

function debounce(fn, time) {
  let timer = null 
  return function (...args) {
    if (timer) clearTimeout(timer)
    timer = setTimeout(() => {
      fn.apply(this, args)
    }, time)
  }
}

function myReduce(func, initVal) {
  let arr = this 
  let base = initVal ? initVal : arr[0]
  let startPoint = initVal ? 1 : 0 
  arr.slice(startPoint).forEach((val, index) => {
    let base = func(base, val, index+startPoint, arr)
  })
  return base
}
